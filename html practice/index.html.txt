<script>
  /* ── Floating hearts ── */
  const bgEl = document.getElementById('heartsBg');
  ['💕','🌸','✨','🎀','💖','🌷','🍬','⭐','💗'].forEach(e => {
    for (let i = 0; i < 4; i++) {
      const h = document.createElement('div');
      h.className = 'heart-bg'; h.textContent = e;
      h.style.cssText = `left:${Math.random()*100}%;--s:${(Math.random()*1.2+0.8).toFixed(1)}rem;--d:${(Math.random()*7+6).toFixed(1)}s;--dl:-${(Math.random()*10).toFixed(1)}s;`;
      bgEl.appendChild(h);
    }
  });

  /* ══════════════════════════════
     GALLERY — unlimited photos
  ══════════════════════════════ */
  const grid = document.getElementById('galleryGrid');
  const countBadge = document.getElementById('countBadge');
  let photos = [];   // { src: dataURL }
  let lbIdx = 0;

  // Always keep exactly one empty slot at the end (the "add" slot)
  function render() {
    grid.innerHTML = '';
    photos.forEach((p, i) => grid.appendChild(makeFilledSlot(i, p.src)));
    grid.appendChild(makeEmptySlot());
    updateCount();
  }

  function makeFilledSlot(index, src) {
    const slot = document.createElement('div');
    slot.className = 'photo-slot filled';
    slot.innerHTML = `
      <img class="loaded" src="${src}" alt="photo ${index+1}"/>
      <button class="remove-btn" title="Remove">✕</button>
    `;
    slot.querySelector('img').addEventListener('click', () => openLb(index));
    slot.querySelector('.remove-btn').addEventListener('click', e => {
      e.stopPropagation();
      photos.splice(index, 1);
      render();
    });
    return slot;
  }

  function makeEmptySlot() {
    const slot = document.createElement('div');
    slot.className = 'photo-slot';
    slot.innerHTML = `
      <div class="slot-placeholder">
        <span class="slot-icon">＋</span>
        <div>Add Photo</div>
      </div>
    `;
    slot.addEventListener('click', () => document.getElementById('photoInput').click());
    return slot;
  }

  function updateCount() {
    const n = photos.length;
    countBadge.textContent = n === 0 ? '0 photos' : `${n} photo${n>1?'s':''}`;
    // auto-scroll to bottom after adding
  }

  function addFromFiles(files) {
    const arr = Array.from(files);
    let loaded = 0;
    arr.forEach(file => {
      const r = new FileReader();
      r.onload = e => {
        photos.push({ src: e.target.result });
        loaded++;
        if (loaded === arr.length) {
          render();
          // scroll gallery to bottom to show new photos
          const scroll = document.getElementById('galleryScroll');
          scroll.scrollTop = scroll.scrollHeight;
        }
      };
      r.readAsDataURL(file);
    });
  }

  document.getElementById('photoInput').addEventListener('change', function() {
    if (this.files.length) { addFromFiles(this.files); this.value = ''; }
  });
  document.getElementById('photoInputBulk').addEventListener('change', function() {
    if (this.files.length) { addFromFiles(this.files); this.value = ''; }
  });

  /* ── Lightbox ── */
  function openLb(index) {
    lbIdx = index;
    showLb();
    document.getElementById('lightbox').classList.add('open');
  }
  function showLb() {
    document.getElementById('lbImg').src = photos[lbIdx].src;
    document.getElementById('lbCounter').textContent = `${lbIdx+1} / ${photos.length}`;
  }
  function closeLb() { document.getElementById('lightbox').classList.remove('open'); }
  function lbNav(dir) {
    lbIdx = (lbIdx + dir + photos.length) % photos.length;
    showLb();
  }
  document.getElementById('lightbox').addEventListener('click', e => {
    if (e.target === document.getElementById('lightbox')) closeLb();
  });
  document.addEventListener('keydown', e => {
    if (!document.getElementById('lightbox').classList.contains('open')) return;
    if (e.key === 'ArrowLeft')  lbNav(-1);
    if (e.key === 'ArrowRight') lbNav(1);
    if (e.key === 'Escape') closeLb();
  });

  /* ── Music ── */
  const audio = document.getElementById('audioEl');
  const playBtn = document.getElementById('playBtn');
  const allBars = document.querySelectorAll('.bar');
  let playing = false;

  document.getElementById('musicInput').addEventListener('change', function() {
    if (!this.files[0]) return;
    audio.src = URL.createObjectURL(this.files[0]);
    audio.volume = parseFloat(document.getElementById('volSlider').value);
    document.getElementById('trackName').textContent = this.files[0].name.replace(/\.[^.]+$/, '');
    document.getElementById('player').classList.add('visible');
    audio.play(); setPlaying(true);
  });
  document.getElementById('volSlider').addEventListener('input', function() { audio.volume = parseFloat(this.value); });
  audio.addEventListener('ended', () => setPlaying(false));

  function togglePlay() { playing ? (audio.pause(), setPlaying(false)) : (audio.play(), setPlaying(true)); }
  function setPlaying(s) {
    playing = s; playBtn.textContent = s ? '⏸' : '▶';
    allBars.forEach(b => s ? b.classList.add('playing') : b.classList.remove('playing'));
  }
